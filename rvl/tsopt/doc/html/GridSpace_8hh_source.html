192  template<typename T> T** dimView(T* x, size_t n, size_t chunklen) {
193  size_t nchunks = n/chunklen;
194  if (n != nchunks*chunklen) {
195  RVLException e;
196  e<<"Error: dimView\n";
197  e<<" length of input array = "<<n<<" not divisible by chunk length = "
198  <<chunklen<<"\n";
199  throw e;
200  }
201  T ** xchunks = (T**)usermalloc_(nchunks*sizeof(T*));
202  for (size_t i=0; i< nchunks; i++) xchunks[i] = &(x[i*chunklen]);
203  return xchunks;
204  }
